import { useSelector, useDispatch } from 'react-redux'
import LiveDemo from 'demos/live-demo/LiveDemo'

Well, isn't this great: [react-redux](https://react-redux.js.org/) now provides a [hooks](https://react-redux.js.org/api/hooks) API! 
In my front-end development work, I've probably overused redux. In many cases, 
using React's Context API provides a simpler and cleaner way of getting state to components that need it, especially when 
combining context and reducer hooks. Of course, it's fine to use both approaches in a single application; which one you use in any particular situation may depend on 
whether your requirements call for the extra flexibility and extendability (custom middleware, for example) of redux.

When using redux, the new hooks API provides a convenient way to access state in and dispatch actions from functional components. 

<LiveDemo
  noInline={true}
  scope={{ useSelector, useDispatch }}
  code={`
const Form = () => {
  const name = useSelector(state => state.name)
  const dispatch = useDispatch()
  
  const onChange = event => {
    dispatch({ type: 'NAME_SET', name: event.target.value })
  }
  
  return (
    <form>
      <input 
        type="text"
        placeholder="Name"
        value={name}
        onChange={onChange} />
    </form>
  )
}
  
const Name = () => { 
  const name = useSelector(state => state.name)
  return (
    <div>
      Name: {name}
    </div>
  )
} 
 
render(
  <React.Fragment>
    <Form />
    <Name />
  </React.Fragment>
)
`}
/>
